\begin{enumerate}[1.]
\item The set $2\Z := \SET{2n:n\in\Z} = \SET{0, \pm 2, \pm 4,\ldots}$ of even integers is a binary algebraic structure under addition.  Prove that $(\Z,+)\cong (2\Z,+)$.
\item For $n\in\Z^{+}$ with $n > 1$, let
\begin{center}
$U_{n} = \SET{z\in\C:z^n = 1}$
\end{center}
be the set of all $n$th roots of unity.  Prove that $(U_{n},\cdot)$ is isomorphic to $(\overline{Z}_{n}, +_{n})$.
\end{enumerate}
